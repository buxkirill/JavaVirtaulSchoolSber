    public static void main(String[] args) {
        BigInteger two = new BigInteger("2");

        for (int p = 0; p <= 100; p++) {
            if (isPrime(p)) {
                System.out.println(two.pow(p).subtract(BigInteger.valueOf(1)));
            }
        }

    }

    public static boolean isPrime(int value) {
        for (int i = 2; i <= value / 2; ++i) {
            if (value % i == 0) {
                return false;
            }
        }
        return true;
    }
